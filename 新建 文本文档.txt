def auto_(rects):
    for rect in rects:
        height = rect.get_height()
        if height < 0:
            plt.text(rect.get_x()+rect.get_width()/2., 1.02*height, f'{height:.2%}', ha='center', va='top', size=10)
        else:
            plt.text(rect.get_x()+rect.get_width()/2., 1.02*height, f'{height:.2%}', ha='center', va='bottom', size=10)
